modals.html

<!-- Start component HTML -->
<a href="#" class="message">Message modal</a>
<a href="#" class="interstitial">Interstitial modal</a>
<a href="#" class="smaller">Small modal</a>
<div id="interstitial" class="modalContainer">
  <div class="modal">
    <span class="close"><i class="icons8-delete-2"></i></span>
    <h3>This is an interstitial</h3>
    <p>If you need to give the user a temporary message that will go away, use this. This message will go away after 15 seconds. Using a loader/spinner here is also a good idea. Having a cancel or alternative CTA is smart here as well.</p>
    <div class="actionContainer">
      <button class="button button--medium button--hover-back button--left button--cancel">Cancel</button>
    </div>
  </div>
</div>
<div id="message" class="modalContainer">
  <div class="modal">
    <span class="close"><i class="icons8-delete-2"></i></span>
    <h3>Basic message modal</h3>
    <p>This modal would be used if there is an imperative message that is required to be seen by the user. It should be revealed based on a users action only. Ex. Displaying an alert because a user made an error and cannot 'proceed'. More than likely we would use a CTA.</p>
    <div class="actionContainer">
      <button class="button button--medium button--hover-back button--left button--cancel">Cancel</button><button class="button button--medium button--hover-go button--right">Go for it</button>
    </div>
  </div>
</div>
<div id="smaller" class="modalContainer">
  <div class="modal smallerModal">
    <span class="close"><i class="icons8-delete-2"></i></span>
    <h3>Basic message modal (smaller)</h3>
    <p>This is the same as the basic modal except a little smaller. Just add a class="smaller" and you can keep the size down if you have less content.</p>
    <div class="actionContainer">
      <button class="button button--medium button--hover-back button--left button--cancel">Cancel</button><button class="button button--medium button--hover-go button--right">Go for it</button>
    </div>
  </div>
</div>
<!-- End component HTML -->

modals.css

@import url('buttons.css');

.modalContainer{
  position: fixed;
  top: 0;
  height: 100vh;
  width: 100vw;
  background: rgba(0, 0, 0, 0.5);
  display: none;
}
.modal{
  border-radius: var(--border-radius);
  height: auto;
  width: auto;
  max-width: 600px;
  margin: 10vh auto;
  padding: var(--m) var(--l);
  background-color: #fff;
  box-shadow: 0 2px 2px 0 rgba(0,0,0,0.14),0 1px 5px 0 rgba(0,0,0,0.12),0 3px 1px -2px rgba(0,0,0,0.2);
}
.modal h3{
  text-transform: uppercase;
  font-weight: 800;
}
.modal button{
  margin: 10px;
}
.close{
  position: absolute;
  color: #333;
  margin-left: calc(600px - 2 * var(--l));
  padding: 5px;
  margin-top: -10px;
  cursor: pointer;
}
.actionContainer{
  text-align: center;
}
.smallerModal{
  max-width: 400px;
}
.smallerModal .close{
  margin-left: calc(400px - 2 * var(--l));
}
@media (max-width: 900px){
  .modal{
    max-width: 80vw;
  }
  .close{
    margin-left: calc(80vw - 2 * var(--l));
  }
  .smallerModal .close{
    margin-left: calc(80vw - 2 * var(--l));
  }
}

modals.js

$('.interstitial').click(function(){
  $('#interstitial').show().delay(15000).fadeOut();;
});
$('.message').click(function(){
  $('#message').show();
});
$('.smaller').click(function(){
  $('#smaller').show();
});
$('.close, .button--cancel').click(function(){
  $(this).parents('.modalContainer').hide();
});
